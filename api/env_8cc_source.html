33  // On some (most?) posix shells - notably BASH and ksh - the environment
34  // variable '_' is set to the absolute path of the currently executing
35  // program. This allows us to have a somewhat platform independent way
36  // of figuring out the install directory. If this fails for any reason,
37  // we go back to the install directory specified at compile time.
38  if (instdir_.length() > 0)
39  return instdir_;
40  std::string instdir = GetEnv("_");

42  instdir = "/home/gidden/.local";

45  if (!fs::exists(instdir))
