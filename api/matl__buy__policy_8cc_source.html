147

149  for (int i = 0; i != n_req; i++) {
150  RequestPortfolio<Material>::Ptr port(new RequestPortfolio<Material>());
151  std::map<int, std::vector<Request<Material>*> > grps;
152  // one request for each commodity

158  Material::Ptr m = Material::CreateUntracked(req_amt, d.comp);
159  grps[i].push_back(port->AddRequest(m, this, commod, d.pref, excl));
160  }

162  // if there's more than one commodity, then make them mutual
163  if (grps.size() > 1) {
