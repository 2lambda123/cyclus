00293   conds.push_back(Cond("EnterTime", "<=", t_));

00300       continue;

00303     std::vector<Cond> conds;
00304     conds.push_back(Cond("AgentId", "==", id));
00305     conds.push_back(Cond("ExitTime", "<", t_));
00306     try {
00307       QueryResult qexit = b_->Query("AgentExit", &conds);
00308       if (qexit.rows.size() != 0) {
00309         continue;  // agent was decomissioned before t_ - skip
00310       }
00311     } catch (std::exception err) {}  // table doesn't exist (okay)
00312 
00313     // if the agent wasn't decommissioned before t_ create and init it
00314 
00315     std::string proto = qentry.GetVal<std::string>("Prototype", i);
00316     std::string impl = qentry.GetVal<std::string>("Spec", i);
00317     AgentSpec spec(impl);
00318     Agent* m = DynamicModule::Make(ctx_, spec);
00319 
00320     // agent-kernel init
00321     m->prototype_ = proto;
00322     m->id_ = id;
00323     m->enter_time_ = qentry.GetVal<int>("EnterTime", i);
00324     unbuilt[id] = m;
00325     parentmap[id] = qentry.GetVal<int>("ParentId", i);
00326 
00327     // agent-custom init
00328     conds.pop_back();
00329     conds.push_back(Cond("SimTime", "==", t_));
00330     CondInjector ci(b_, conds);
00331     PrefixInjector pi(&ci, "AgentState");
00332     m->Agent::InitFrom(&pi);
00333     pi = PrefixInjector(&ci, "AgentState" + spec.Sanitize());

00335   }
00336 
00337   // construct agent hierarchy starting at roots (no parent) down
00338   std::map<int, Agent*>::iterator it = unbuilt.begin();
00339   std::vector<Agent*> enter_list;
00340   while (unbuilt.size() > 0) {
00341     int id = it->first;
00342     Agent* m = it->second;
00343     int parentid = parentmap[id];
00344 
00345     if (parentid == -1) {  // root agent
00346       m->Connect(NULL);
00347       agents_[id] = m;
00348       ++it;
00349       unbuilt.erase(id);
00350       enter_list.push_back(m);
00351     } else if (agents_.count(parentid) > 0) {  // parent is built
00352       m->Connect(agents_[parentid]);
00353       agents_[id] = m;
00354       ++it;
00355       unbuilt.erase(id);
00356       enter_list.push_back(m);
00357     } else {  // parent not built yet
00358       ++it;
00359     }
00360     if (it == unbuilt.end()) {
00361       it = unbuilt.begin();
00362     }
00363   }
00364 
00365   // notify all agents that they are active in a simulation AFTER the
00366   // parent-child hierarchy has been reconstructed.
00367   for (int i = 0; i < enter_list.size(); ++i) {
00368     enter_list[i]->EnterNotify();
