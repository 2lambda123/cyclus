1854 void Hdf5Back::FillBuf(std::string title, char* buf, DatumList& group,
1855  size_t* sizes, size_t rowsize) {

1857  using std::string;
1858  using std::vector;
1859  using std::set;
1860  using std::list;
1861  using std::pair;
1862  using std::map;
1863  Datum::Vals vals;
1864  Datum::Shape shape;
1865  Datum::Shapes shapes;
1866  Datum::Vals header = group.front()->vals();
1867  int ncols = header.size();

1869 
1870  size_t offset = 0;
1871  const void* val;
1872  size_t fieldlen;
1873  size_t valuelen;
1874  DatumList::iterator it;
1875  for (it = group.begin(); it != group.end(); ++it) {
1876  vals = (*it)->vals();
1877  shapes = (*it)->shapes();
1878  for (int col = 0; col < ncols; ++col) {
1879  const boost::spirit::hold_any* a = &(vals[col].second);
1880  switch (dbtypes[col]) {
1881  case BOOL:
1882  case INT:
1883  case FLOAT:
1884  case DOUBLE: {
1885  val = a->castsmallvoid();
1886  memcpy(buf + offset, val, sizes[col]);
1887  break;
1888  }
1889  case STRING: {
1890  const std::string s = a->cast<std::string>();
1891  fieldlen = sizes[col];
1892  valuelen = std::min(s.size(), fieldlen);
1893  memcpy(buf + offset, s.c_str(), valuelen);
1894  memset(buf + offset + valuelen, 0, fieldlen - valuelen);
1895  break;
1896  }
1897  case VL_STRING: {
1898  Digest key = VLWrite<std::string, VL_STRING>(a);
1899  memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
1900  break;
1901  }
1902  case BLOB: {
1903  Digest key = VLWrite<Blob, BLOB>(a);
1904  memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
1905  break;
1906  }
1907  case UUID: {
1908  boost::uuids::uuid uuid = a->cast<boost::uuids::uuid>();
1909  memcpy(buf + offset, &uuid, CYCLUS_UUID_SIZE);
1910  break;
1911  }
1912  case VECTOR_INT: {
1913  std::vector<int> val = a->cast<std::vector<int> >();
1914  fieldlen = sizes[col];
1915  valuelen = std::min(val.size() * sizeof(int), fieldlen);
1916  memcpy(buf + offset, &val[0], valuelen);
1917  memset(buf + offset + valuelen, 0, fieldlen - valuelen);
1918  break;
1919  }
1920  case VL_VECTOR_INT: {
1921  Digest key = VLWrite<std::vector<int>, VL_VECTOR_INT>(a);
1922  memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
1923  break;
1924  }
1925  case VECTOR_FLOAT: {
1926  std::vector<float> val = a->cast<std::vector<float> >();
1927  fieldlen = sizes[col];
1928  valuelen = std::min(val.size() * sizeof(float), fieldlen);
1929  memcpy(buf + offset, &val[0], valuelen);
1930  memset(buf + offset + valuelen, 0, fieldlen - valuelen);
1931  break;
1932  }
1933  case VL_VECTOR_FLOAT: {
1934  Digest key = VLWrite<std::vector<float>, VL_VECTOR_FLOAT>(a);
1935  memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
1936  break;
1937  }
1938  case VECTOR_DOUBLE: {
1939  std::vector<double> val = a->cast<std::vector<double> >();
1940  fieldlen = sizes[col];
1941  valuelen = std::min(val.size() * sizeof(double), fieldlen);
1942  memcpy(buf + offset, &val[0], valuelen);
1943  memset(buf + offset + valuelen, 0, fieldlen - valuelen);
1944  break;
1945  }
1946  case VL_VECTOR_DOUBLE: {
1947  Digest key = VLWrite<std::vector<double>, VL_VECTOR_DOUBLE>(a);
1948  memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
1949  break;
1950  }
1951  case VECTOR_STRING: {
1952  vector<string> val = a->cast<vector<string> >();
1953  shape = shapes[col];
1954  fieldlen = shape[1];
1955  unsigned int cnt = 0;
1956  for (; cnt < val.size(); ++cnt) {
1957  valuelen = std::min(val[cnt].size(), fieldlen);
1958  memcpy(buf + offset + fieldlen*cnt, val[cnt].c_str(), valuelen);
1959  memset(buf + offset + fieldlen*cnt + valuelen, 0, fieldlen - valuelen);
1960  }
1961  memset(buf + offset + fieldlen*cnt, 0, fieldlen * (shape[0] - cnt));
1962  break;
1963  }
1964  case VECTOR_VL_STRING: {
1965  vector<string> val = a->cast<vector<string> >();
