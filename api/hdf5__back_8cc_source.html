01854 void Hdf5Back::FillBuf(std::string title, char* buf, DatumList& group,
01855                        size_t* sizes, size_t rowsize) {
01856   using std::min;
01857   using std::string;
01858   using std::vector;
01859   using std::set;
01860   using std::list;
01861   using std::pair;
01862   using std::map;
01863   Datum::Vals vals;
01864   Datum::Shape shape;
01865   Datum::Shapes shapes;
01866   Datum::Vals header = group.front()->vals();
01867   int ncols = header.size();
01868   DbTypes* dbtypes = schemas_[title];
01869 
01870   size_t offset = 0;
01871   const void* val;
01872   size_t fieldlen;
01873   size_t valuelen;
01874   DatumList::iterator it;
01875   for (it = group.begin(); it != group.end(); ++it) {
01876     vals = (*it)->vals();
01877     shapes = (*it)->shapes();
01878     for (int col = 0; col < ncols; ++col) {
01879       const boost::spirit::hold_any* a = &(vals[col].second);
01880       switch (dbtypes[col]) {
01881         case BOOL:
01882         case INT:
01883         case FLOAT:
01884         case DOUBLE: {
01885           val = a->castsmallvoid();
01886           memcpy(buf + offset, val, sizes[col]);
01887           break;
01888         }
01889         case STRING: {
01890           const std::string s = a->cast<std::string>();
01891           fieldlen = sizes[col];
01892           valuelen = std::min(s.size(), fieldlen);
01893           memcpy(buf + offset, s.c_str(), valuelen);
01894           memset(buf + offset + valuelen, 0, fieldlen - valuelen);
01895           break;
01896         }
01897         case VL_STRING: {
01898           Digest key = VLWrite<std::string, VL_STRING>(a);
01899           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
01900           break;
01901         }
01902         case BLOB: {
01903           Digest key = VLWrite<Blob, BLOB>(a);
01904           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
01905           break;

02397           Digest key = VLWrite<map<string, double>, VL_MAP_VL_STRING_DOUBLE>(a);
02398           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
02399           break;
02400         }
02401         case MAP_VL_STRING_STRING: {
02402           map<string, string> val = a->cast<map<string, string> >();
02403           Digest keyhash;
02404           shape = shapes[col];
02405           size_t vallen = shape[2];
02406           fieldlen = CYCLUS_SHA1_SIZE + vallen;
02407           int truevallen;
02408           unsigned int cnt = 0;
02409           map<string, string>::iterator valit = val.begin();
02410           for (; valit != val.end(); ++valit) {
02411             keyhash = VLWrite<string, VL_STRING>(valit->first);
02412             memcpy(buf + offset + fieldlen*cnt, keyhash.val, CYCLUS_SHA1_SIZE);
02413             truevallen = std::min(valit->second.size(), vallen);
02414             memcpy(buf + offset + fieldlen*cnt + CYCLUS_SHA1_SIZE, valit->second.c_str(), truevallen);
02415             memset(buf + offset + fieldlen*cnt + CYCLUS_SHA1_SIZE + truevallen, 0,
02416                    vallen - truevallen);
02417             ++cnt;
02418           }
02419           memset(buf + offset + fieldlen*cnt, 0, fieldlen * (val.size() - cnt));
02420           break;
02421         }
02422         case VL_MAP_VL_STRING_STRING: {
02423           Digest key = VLWrite<map<string, string>, VL_MAP_VL_STRING_STRING>(a);
02424           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
02425           break;
02426         }
02427         case MAP_VL_STRING_VL_STRING: {
02428           map<string, string> val = a->cast<map<string, string> >();
02429           Digest keyhash;
02430           Digest valhash;
02431           shape = shapes[col];
02432           fieldlen = 2 * CYCLUS_SHA1_SIZE;
02433           unsigned int cnt = 0;
02434           map<string, string>::iterator valit = val.begin();
02435           for (; valit != val.end(); ++valit) {
02436             keyhash = VLWrite<string, VL_STRING>(valit->first);
02437             memcpy(buf + offset + fieldlen*cnt, keyhash.val, CYCLUS_SHA1_SIZE);
02438             valhash = VLWrite<string, VL_STRING>(valit->second);
02439             memcpy(buf + offset + fieldlen*cnt + CYCLUS_SHA1_SIZE, valhash.val,
02440                    CYCLUS_SHA1_SIZE);
02441             ++cnt;
02442           }
02443           memset(buf + offset + fieldlen*cnt, 0, fieldlen * (val.size() - cnt));
02444           break;
02445         }
02446         case VL_MAP_VL_STRING_VL_STRING: {
02447           Digest key = VLWrite<map<string, string>, VL_MAP_VL_STRING_VL_STRING>(a);
02448           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
02449           break;
02450         }
02451         case MAP_PAIR_INT_STRING_DOUBLE: {
02452           map<pair<int, string>, double> val = \
02453             a->cast<map<pair<int, string>, double> >();
02454           shape = shapes[col];
02455           int strlen = shape[1];
02456           fieldlen = sizeof(int) + strlen + sizeof(double);
02457           unsigned int cnt = 0;
02458           map<pair<int, string>, double>::iterator valit = val.begin();
02459           for (; valit != val.end(); ++valit) {
02460             valuelen = std::min(static_cast<int>(valit->first.second.size()), strlen);
02461             memcpy(buf + offset + fieldlen*cnt, &(valit->first.first), sizeof(int));
02462             memcpy(buf + offset + fieldlen*cnt + sizeof(int), 
02463                    valit->first.second.c_str(), valuelen);
02464             memset(buf + offset + fieldlen*cnt + sizeof(int) + valuelen, 0, 
02465                    strlen - valuelen);
02466             memcpy(buf + offset + fieldlen*cnt + sizeof(int) + strlen, &(valit->second),
02467                    sizeof(double));
02468             ++cnt;
02469           }
02470           memset(buf + offset + fieldlen*cnt, 0, fieldlen * (shape[0] - cnt));
02471           break;
02472         }
02473         case VL_MAP_PAIR_INT_STRING_DOUBLE: {
02474           shape = shapes[col];
02475           size_t strlen = shape[1];
02476           map<pair<int, string>, double> givenval = \
02477             a->cast<map<pair<int, string>, double> >();
02478           map<pair<int, string>, double> val;
02479           // ensure string is of specified length
02480           map<pair<int, string>, double>::iterator valit = givenval.begin();
02481           for (; valit != givenval.end(); ++valit)
02482             val[std::make_pair(valit->first.first, 
02483                 string(valit->first.second, 0, strlen))] = valit->second;
02484           Digest key = VLWrite<map<pair<int, string>, double>, 
02485                                VL_MAP_PAIR_INT_STRING_DOUBLE>(val);
02486           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
02487           break;
02488         }
02489         case MAP_PAIR_INT_VL_STRING_DOUBLE: {
02490           map<pair<int, string>, double> val = \
02491             a->cast<map<pair<int, string>, double> >();
02492           Digest keyhash;
02493           fieldlen = sizeof(int) + CYCLUS_SHA1_SIZE + sizeof(double);
02494           unsigned int cnt = 0;
02495           map<pair<int, string>, double>::iterator valit = val.begin();
02496           for (; valit != val.end(); ++valit) {
02497             keyhash = VLWrite<string, VL_STRING>(valit->first.second);
02498             memcpy(buf + offset + fieldlen*cnt, &(valit->first.first), sizeof(int));
02499             memcpy(buf + offset + fieldlen*cnt + sizeof(int), keyhash.val, 
02500                    CYCLUS_SHA1_SIZE);
02501             memcpy(buf + offset + fieldlen*cnt + sizeof(int) + CYCLUS_SHA1_SIZE, 
02502                    &(valit->second), sizeof(double));
02503             ++cnt;
02504           }
02505           memset(buf + offset + fieldlen*cnt, 0, fieldlen * (val.size() - cnt));
02506           break;
02507         }
02508         case VL_MAP_PAIR_INT_VL_STRING_DOUBLE: {
02509           Digest key = VLWrite<map<pair<int, string>, double>, 
02510                                VL_MAP_PAIR_INT_VL_STRING_DOUBLE>(a);
02511           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
02512           break;
02513         }
02514         default: {
02515           throw ValueError("attempted to retrieve unsupported HDF5 backend type");
02516         }
02517       }
02518       offset += sizes[col];
02519     }
02520   }
02521 }

02523 template <typename T, DbTypes U>
02524 T Hdf5Back::VLRead(const char* rawkey) {
02525   // key is used as offset
02526   Digest key;
02527   memcpy(key.val, rawkey, CYCLUS_SHA1_SIZE);
02528   const std::vector<hsize_t> idx = key.cast<hsize_t>();
02529   hid_t dset = VLDataset(U, false);
02530   hid_t dspace = H5Dget_space(dset);
02531   hid_t mspace = H5Screate_simple(CYCLUS_SHA1_NINT, vlchunk_, NULL);
02532   herr_t status = H5Sselect_hyperslab(dspace, H5S_SELECT_SET, (const hsize_t*) &idx[0],
02533                                       NULL, vlchunk_, NULL);
02534   if (status < 0)
02535     throw IOError("could not select hyperslab of value array for reading "
02536                   "in the database '" + path_ + "'.");
02537   hvl_t buf;
02538   status = H5Dread(dset, vldts_[U], mspace, dspace, H5P_DEFAULT, &buf);
02539   if (status < 0) {
02540     std::stringstream ss;
02541     ss << U;
02542     throw IOError("failed to read in variable length data "
02543                   "in the database '" + path_ + "' (type id " + ss.str() +
02544                   ").");
02545   }
02546   T val = VLBufToVal<T>(buf);
02547   status = H5Dvlen_reclaim(vldts_[U], mspace, H5P_DEFAULT, &buf);
02548   if (status < 0)
02549     throw IOError("failed to reclaim variable length data space "
02550                   "in the database '" + path_ + "'.");
02551   H5Sclose(mspace);
02552   H5Sclose(dspace);
02553   return val;
02554 }

02726   if (status < 0)

02728                   "in the database '" + path_ + "'.");

02730   H5Sclose(dspace);
