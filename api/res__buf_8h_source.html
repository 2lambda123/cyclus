00025 /// ResBuf is a helper class that provides semi-automated management of
00026 /// a collection of resources (e.g. agent stocks and inventories).
00027 /// Constructed buffers have infinite capacity unless explicitly changed.
00028 /// Resource popping occurs in the order the resources were pushed (i.e. oldest
00029 /// resources are popped first), unless explicitly specified otherwise.
00030 ///
00031 /// Typically, a ResBuf will be a member variable on an agent/archetype class.
00032 /// Resources can be added and retrieved from it as needed, and the buffer can
00033 /// be queried in various ways as done in the example below:
00034 ///
00035 /// @code
00036 /// class MyAgent : public cyclus::Facility {
00037 ///  public:
00038 ///   Tick() {
00039 ///     double batch_size = 2703;
00040 ///     if (outventory_.space() < batch_size) {
00041 ///       return;
00042 ///     } else if (inventory_.quantity() < batch_size) {
00043 ///       return;
00044 ///     }
00045 ///
00046 ///     outventory_.Push(inventory_.Pop(batch_size));
00047 ///   }
00048 ///
00049 ///   ... // resource exchange to fill up inventory_ buffer
00050 ///
00051 ///  private:
00052 ///   ...
00053 ///   cyclus::toolkit::ResBuf<cyclus::Material> inventory_;
00054 ///   cyclus::toolkit::ResBuf<cyclus::Material> outventory_;
00055 /// };
00056 /// @endcode
