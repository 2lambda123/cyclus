25 /// ResBuf is a helper class that provides semi-automated management of
26 /// a collection of resources (e.g. agent stocks and inventories).
27 /// Constructed buffers have infinite capacity unless explicitly changed.
28 /// Resource popping occurs in the order the resources were pushed (i.e. oldest
29 /// resources are popped first), unless explicitly specified otherwise.
30 ///
31 /// Typically, a ResBuf will be a member variable on an agent/archetype class.
32 /// Resources can be added and retrieved from it as needed, and the buffer can
33 /// be queried in various ways as done in the example below:
34 ///
35 /// @code
36 /// class MyAgent : public cyclus::Facility {
37 /// public:
38 /// Tick() {
39 /// double batch_size = 2703;
40 /// if (outventory_.space() < batch_size) {
41 /// return;
42 /// } else if (inventory_.quantity() < batch_size) {
43 /// return;
44 /// }
45 ///
46 /// outventory_.Push(inventory_.Pop(batch_size));
47 /// }
48 ///
49 /// ... // resource exchange to fill up inventory_ buffer
50 ///
51 /// private:
52 /// ...
53 /// cyclus::toolkit::ResBuf<cyclus::Material> inventory_;
54 /// cyclus::toolkit::ResBuf<cyclus::Material> outventory_;
55 /// };
56 /// @endcode
57 ///
58 /// In this example, if there is sufficient material in inventory_, 2703 kg is
59 /// removed as a single object that is then placed in another buffer
60 /// (outventory_) each time step.

108  /// resource object.
109  /// Resources are split if necessary in order to pop the exact quantity

112  /// when returned.

239  /// Pushes a single resource object to the buffer.
240  /// Resource objects are never combined in the buffer; they are stored as
241  /// unique objects. The resource object is only pushed to the buffer if it
242  /// does not cause the buffer to exceed its capacity.
243  ///
244  /// @throws ValueError the pushing of the given resource object would
245  /// cause the buffer to exceed its capacity.
246  ///
247  /// @throws KeyError the resource object to be pushed is already present
248  /// in the buffer.
249  void Push(Resource::Ptr r) {
250  typename T::Ptr m = boost::dynamic_pointer_cast<T>(r);
251  if (m == NULL) {
252  throw CastError("pushing wrong type of resource onto ResBuf");
253  } else if (r->quantity() - space() > eps_rsrc()) {
254  std::stringstream ss;
255  ss << "resource pushing breaks capacity limit: space=" << space()
256  << ", rsrc->quantity()=" << r->quantity();
257  throw ValueError(ss.str());
258  } else if (rs_present_.count(m) == 1) {
259  throw KeyError("duplicate resource push attempted");
260  }
261 
262  rs_.push_back(m);
263  rs_present_.insert(m);
264  qty_ += r->quantity();
265  UpdateQty();
266  }

268  /// Pushes one or more resource objects (as a std::vector) to the buffer.
269  /// Resource objects are never squashed in the buffer; they are stored as
270  /// unique objects. The resource objects are only pushed to the buffer if
271  /// they do not cause the buffer to exceed its capacity; otherwise none of the
272  /// given resource objects are added to the buffer.
273  ///
274  /// @throws ValueError adding the given resource objects would
275  /// cause the buffer to exceed its capacity.
276  ///
277  /// @throws KeyError one or more of the resource objects to be added
278  /// are already present in the buffer.
279  template <class B>
280  void Push(std::vector<B> rs) {
281  std::vector<typename T::Ptr> rss;
282  typename T::Ptr r;
283  for (int i = 0; i < rs.size(); i++) {
284  r = boost::dynamic_pointer_cast<T>(rs[i]);
285  if (r == NULL) {
286  throw CastError("pushing wrong type of resource onto ResBuf");
287  }
288  rss.push_back(r);
289  }
290

292  for (int i = 0; i < rss.size(); i++) {
293  tot_qty += rss.at(i)->quantity();
294  }
295  if (tot_qty - space() > eps_rsrc()) {
296  throw ValueError("Resource pushing breaks capacity limit.");

298
