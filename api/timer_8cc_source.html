120 void Timer::SchedDecom(Agent* m, int t) {
121  if (t < time_) {
122  throw ValueError("Cannot schedule decommission for t < [current-time]");
123  }
124 
125  // It is possible that a single agent may be scheduled for decommissioning
126  // multiple times. If this happens, we cannot just add it to the queue again
127  // - the duplicate entries will result in a double delete attempt and
128  // segfaults and otherwise bad things. Remove previous decommissionings
129  // before scheduling this new one.
130  std::map<int, std::vector<Agent*> >::iterator it;
131  bool done = false;
132  for (it = decom_queue_.begin(); it != decom_queue_.end(); ++it) {
133  int t = it->first;
134  std::vector<Agent*> ags = it->second;
135  for (int i = 0; i < ags.size(); i++) {
136  if (ags[i] == m) {
137  CLOG(LEV_WARN) << "scheduled over previous decommissioning of " << m->id();
138  decom_queue_[t].erase(decom_queue_[t].begin()+i);
139  done = true;
140  break;
141  }
142  }
143  if (done) {
144  break;
145  }
146  }
147 
148  decom_queue_[t].push_back(m);
149 }
