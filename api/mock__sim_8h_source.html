00072 /// MockSim is a helper for running full simulations entirely in-code to test
00073 /// archetypes/agents without having to deal with input files, output database
00074 /// files, and other pieces of the full Cyclus stack.  This is especially
00075 /// convenient for writing unit-like tests (e.g. using gtest) for your
00076 /// archetype's in-simulation behavior.   Initialize the MockSim indicating the
00077 /// archetype you want to test and the simulation duration.  Then add any
00078 /// number of sources and/or sinks to transact with your agent.  They can have
00079 /// specific recipes (or not) and their deployment and lifetime (before
00080 /// decommissioning) can be specified too.  Here is an example using the
00081 /// agents:Source archetype in Cyclus as the tested agent:
00082 ///
00083 /// @code
00084 ///
00085 /// cyclus::CompMap m;
00086 /// m[922350000] = .05;
00087 /// m[922380000] = .95;
00088 /// cyclus::Composition::Ptr fresh = cyclus::Composition::CreateFromMass(m);
00089 ///
00090 /// std::string config =
00091 ///     "<commod>enriched_u</commod>"
00092 ///     "<recipe_name>fresh_fuel</recipe_name>"
00093 ///     "<capacity>10</capacity>";
00094 ///
00095 /// int dur = 10;
00096 /// cyclus::MockSim sim(cyclus::AgentSpec(":agents:Source"), config, dur);
00097 /// sim.AddSink("enriched_u").Finalize();
00098 /// sim.AddRecipe("fresh_fuel", fresh);
00099 /// int src_id = sim.Run(); // capture the agent ID of the facility being tested
00100 ///
00101 /// @endcode
00102 ///
00103 /// Querying the results can be accomplished by getting a reference to the
00104 /// in-memory database generated.  Not all data that is present in normal
00105 /// full-stack simulations is available.  However, most of the key core tables
00106 /// are fully available.  Namely, the Transactions, Composition, Resources,
00107 /// ResCreators, AgentEntry, and AgentExit tables are available.  Any
00108 /// custom-tables created by the tested archetype will also be available.  Here
00109 /// is a sample query and test you might write if using the googletest
00110 /// framework:
00111 ///
00112 /// @code
00113 ///
00114 /// // return all transactions where our source facility is the sender
00115 /// std::vector<cyclus::Cond> conds;
00116 /// conds.push_back("SenderId", "==", src_id);
00117 /// cyclus::QueryResult qr = sim.db().Query("Transactions", &conds);
00118 /// int n_trans = qr.rows.size();
00119 /// EXPECT_EQ(10, n_trans) << "expected 10 transactions, got " << n_trans;
00120 ///
00121 /// // reconstruct the material object for the first transaction
00122 /// int res_id = qr.GetVal<int>("ResourceId", 0);
00123 /// cyclus::Material::Ptr m = sim.GetMaterial(res_id);
00124 /// EXPECT_DOUBLE_EQ(10, m->quantity());
00125 ///
00126 /// // confirm composition is as expected
00127 /// cyclus::toolkit::MatQuery mq(m);
00128 /// EXPECT_DOUBLE_EQ(0.5, mq.mass(922350000));
00129 /// EXPECT_DOUBLE_EQ(9.5, mq.mass(922380000));
00130 ///
00131 /// @endcode
