92 double GreedySolver::Capacity(ExchangeNode::Ptr n, const Arc& a, bool min_cap,
93  double curr_qty) {
94  if (n->group == NULL) {
95  throw cyclus::StateError("An notion of node capacity requires a nodegroup.");
96  }
97 
98  if (n->unit_capacities[a].size() == 0) {
99  return n->qty - curr_qty;
100  }
101 
102  std::vector<double>& unit_caps = n->unit_capacities[a];
103  const std::vector<double>& group_caps = grp_caps_[n->group];
104  std::vector<double> caps;
105  double grp_cap, u_cap, cap;
106 
107  for (int i = 0; i < unit_caps.size(); i++) {
108  grp_cap = group_caps[i];
109  u_cap = unit_caps[i];
110  cap = grp_cap / u_cap;
111  CLOG(cyclus::LEV_DEBUG1) << "Capacity for node: ";
112  CLOG(cyclus::LEV_DEBUG1) << " group capacity: " << grp_cap;
113  CLOG(cyclus::LEV_DEBUG1) << " unit capacity: " << u_cap;
114  CLOG(cyclus::LEV_DEBUG1) << " capacity: " << cap;
115 
116  // special case for unlimited capacities
117  if (grp_cap == std::numeric_limits<double>::max()) {
118  caps.push_back(std::numeric_limits<double>::max());
119  } else {
120  caps.push_back(cap);
121  }
122  }
123 
124  if (min_cap) { // the smallest value is constraining (for bids)
125  cap = *std::min_element(caps.begin(), caps.end());
126  } else { // the largest value must be met (for requests)
127  cap = *std::max_element(caps.begin(), caps.end());
128  }
